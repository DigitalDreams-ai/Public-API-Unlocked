# Python Example

```python
import requests


def create_submission(base_url: str, api_key: str, payload: dict) -> dict:
    url = f"{base_url}/services/apexrest/v1/publicapi/submissions"
    headers = {
        "Content-Type": "application/json",
        "X-Api-Key": api_key,
    }
    response = requests.post(url, json=payload, headers=headers, timeout=30)
    body = response.json()
    if response.status_code >= 400:
        raise RuntimeError(body.get("errorMessage", f"HTTP {response.status_code}"))
    return body


if __name__ == "__main__":
    result = create_submission(
        "https://your-site.example.com",
        "your-api-key",
        {
            "first_name": "Jane",
            "last_name": "Smith",
            "email": "jane@example.com",
        },
    )
    print(result)
```
